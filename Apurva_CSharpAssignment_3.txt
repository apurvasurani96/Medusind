using System;


public class A
{
      
    public static int count(string s, char c)
    {
        int res = 0;
 
        for (int i = 0; i < s.Length; i++)
        {
            if (s[i] == c)
            res++;
        }
         
        return res;
    }
     
   
    public static void Main()
    {
        string str = "OOPS PROGRAMMING IN C SHARP";
        char c = 'S';
          
         
        Console.WriteLine(count(str, c));
        Console.Read();
    }
}
